import { css } from '@emotion/core'
import { UncontrolledTextareaFieldLight, TextareaFieldLight } from '@csssr/core-design'

# TextareaFieldLight

Компонент, представляющий из себя поле для ввода многострочного текста с текстовой меткой снизу.

<TextareaFieldLight
  id="helloField"
  placeholder="World"
  label="Hello"
  css={css`
    grid-column: 1 / span 6;
  `}
/>

В различных состояниях (uncontrolled):

<UncontrolledTextareaFieldLight label="Empty Field" css={css`grid-column: 1 / span 3;`} value="" placeholder="Empty"/>
<UncontrolledTextareaFieldLight label="Filled Field" css={css`grid-column: 4 / span 3;`} value="Filled"/>
<UncontrolledTextareaFieldLight label="Error Field" css={css`grid-column: 7 / span 3;`} error={true} value="Has Error"/>
<UncontrolledTextareaFieldLight label="Focused Field" css={css`grid-column: 10 / span 3;`} focused={true} value="Focused"/>
<UncontrolledTextareaFieldLight label="Hovered Field" css={css`margin-top: 20px; grid-column: 1 / span 3;`} hovered={true} value="Hovered"/>

---

## Свойства

| Имя   | Допустимые значения | Значение по умолчанию | Комментарий                   |
| ----- | ------------------- | --------------------- | ----------------------------- |
| id    | строка              |                       | Уникальный идентификатор поля |
| error | строка              |                       | Ошибка                        |

---

## Код

```
  import { TextareaFieldLight } from '@csssr/core-design'

  <TextareaFieldLight id="helloField" placeholder="World" label="Hello" />
```
