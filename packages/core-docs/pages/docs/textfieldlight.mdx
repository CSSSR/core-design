import { css } from '@emotion/core'
import { UncontrolledTextFieldLight, TextFieldLight } from '@csssr/core-design'

# TextFieldLight

Компонент, представляющий из себя поле для ввода короткого текста.

<TextFieldLight placeholder="Hello world" id="helloField" label="Hello" css={css`grid-column: 1 / span 6;`}/>

В различных состояниях (uncontrolled):

<UncontrolledTextFieldLight label="Empty Field" value="" placeholder="Empty" css={css`grid-column: 1 / span 3;`} />
<UncontrolledTextFieldLight label="Filled Field" value="Filled" css={css`grid-column: 4 / span 3;`} />
<UncontrolledTextFieldLight label="Error Field" error={true} value="Has Error" css={css`grid-column: 7 / span 3;`} />
<UncontrolledTextFieldLight label="Focused Field" focused={true} value="Focused" css={css`grid-column: 10 / span 3;`} />
<UncontrolledTextFieldLight label="Hovered Field" hovered={true} value="Hovered" css={css`margin-top: 20px; grid-column: 1 / span 3;`} />

---

## Свойства

| Имя   | Допустимые значения | Значение по умолчанию | Комментарий                   |
| ----- | ------------------- | --------------------- | ----------------------------- |
| id    | строка              |                       | Уникальный идентификатор поля |
| error | true \| false       |                       | Ошибка                        |

Можно также передать все нативные аргументы элемента [input](http://htmlbook.ru/html/input).

---------------------------------------
## Код
```js
  import { TextFieldLight } from '@csssr/core-design'

  <TextFieldLight placeholder="Hello world" id="helloField" label="Hello" />
```
