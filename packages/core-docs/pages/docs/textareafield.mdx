import { css } from '@emotion/core'
import { Grid, UncontrolledTextareaFieldLight, TextareaFieldLight } from '@csssr/core-design'

# TextareaFieldLight

Компонент, представляющий из себя поле для ввода многострочного текста c текстовой меткой снизу.

<TextareaFieldLight id="helloField" placeholder="World" label="Hello" />

В различных состояниях (uncontrolled):

<UncontrolledTextareaFieldLight label="Empty Field" css={css`width: 200px`} value="" placeholder="Empty"/>
<UncontrolledTextareaFieldLight label="Filled Field" css={css`width: 200px`} value="Filled"/>
<UncontrolledTextareaFieldLight label="Error Field" css={css`width: 200px`} error={true} value="Has Error"/>
<UncontrolledTextareaFieldLight label="Focused Field" css={css`width: 200px`} focused={true} value="Focused"/>
<UncontrolledTextareaFieldLight label="Hovered Field" css={css`width: 200px`} hovered={true} value="Hovered"/>

---

## Свойства

| Имя   | Допустимые значения | Значение по умолчанию | Комментарий                   |
| ----- | ------------------- | --------------------- | ----------------------------- |
| id    | строка              |                       | Уникальный идентификатор поля |
| error | true \| false       |                       | Ошибка                        |
