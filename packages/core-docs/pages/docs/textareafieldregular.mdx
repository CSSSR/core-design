import { css } from '@emotion/core'
import { TextareaFieldRegular, UncontrolledTextareaFieldRegular } from '@csssr/core-design'

# TextareaFieldRegular

Компонент, представляющий из себя поле для ввода многострочного текста с текстовой меткой снизу.

<TextareaFieldRegular id="helloField" label="Hello" css={css`grid-column: 1 / span 6;`} />

В различных состояниях (uncontrolled):

<UncontrolledTextareaFieldRegular label="Empty Field" css={css`grid-column: 1 / span 3;`} value="" />
<UncontrolledTextareaFieldRegular label="Filled Field" css={css`grid-column: 4 / span 3;`} value="Filled" />
<UncontrolledTextareaFieldRegular label="Error Field" css={css`grid-column: 7 / span 3;`} error={true} value="Has Error" />
<UncontrolledTextareaFieldRegular label="Focused Field" css={css`grid-column: 10 / span 3;`} focused={true} value="Focused" />
<UncontrolledTextareaFieldRegular label="Hovered Field" css={css`margin-top: 20px; grid-column: 1 / span 3;`} hovered={true} value="Hovered" />

---

## Свойства

| Имя   | Допустимые значения | Значение по умолчанию | Комментарий                   |
| ----- | ------------------- | --------------------- | ----------------------------- |
| id    | строка              |                       | Уникальный идентификатор поля |
| error | true \| false       |                       | Ошибка                        |

---------------------------------------
## Код
```js
  import { TextareaFieldRegular } from '@csssr/core-design'

  <TextareaFieldRegular id="helloField" label="Hello" />
```
